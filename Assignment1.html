#Ho = Distribution of bmi is same across women with different children
#Halt = At least one of the bmi distribution is diffferent
#Assuming  = 0.05
 
#Creating a dataframe for processing
df0= pd.DataFrame({'NoOfChildren': '0', 'BMI':children0})
df1 = pd.DataFrame({'NoOfChildren': '1', 'BMI':children1})
df2 = pd.DataFrame({'NoOfChildren': '2', 'BMI':children2})

children_bmi_df= pd.DataFrame()
children_bmi_df = children_bmi_df.append(df0)
children_bmi_df = children_bmi_df.append(df1)
children_bmi_df = children_bmi_df.append(df2)

mod = ols('BMI ~ NoOfChildren', data = children_bmi_df).fit()
aov_table = sm.stats.anova_lm(mod, typ=2)
print(aov_table)

#According to the table, we get value of p is less than that of expected level of signifance at 5%
#So we reject the null hypothesis, the distribution is not the same among women
